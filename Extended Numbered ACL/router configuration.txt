Dhaka Router

Router>enable
Router#configure terminal
Router(config)#hostname Dhaka

Dhaka(config)#interface gig 0/0
Dhaka(config-if)#ip address 192.168.1.1 255.255.255.0
Dhaka(config-if)#no shutdown

Dhaka(config-if)#interface gig 0/1
Dhaka(config-if)#ip address 193.168.1.2 255.255.255.0
Dhaka(config-if)#no shutdown

Dhaka(config-if)#exit

Dhaka(config)#router rip
Dhaka(config-router)#network 192.168.1.0
Dhaka(config-router)#network 193.168.1.0
Dhaka(config-router)#exit
Dhaka(config)#do wr

Dhaka(config)#access-list 110 deny tcp host 192.168.1.2 host 200.168.1.100 eq 80
Dhaka(config)#access-list 110 permit tcp host 192.168.1.2 host 198.168.1.100 eq 80
Dhaka(config)#access-list 110 deny tcp host 192.168.1.150 host 198.168.1.100 eq 80
Dhaka(config)#access-list 110 permit tcp host 192.168.1.65 host 200.168.1.100 eq 80
Dhaka(config)#access-list 110 permit ip any any

Dhaka(config)#interface gig 0/0
Dhaka(config-if)#ip access-group 110 in
Dhaka(config-if)#exit
Dhaka(config)#do wr
