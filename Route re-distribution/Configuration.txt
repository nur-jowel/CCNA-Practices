Router 4 (RIP Routing)


Router>enable
Router#configure terminal

Router(config)#hostname RIP
RIP(config)#interface gig 0/0
RIP(config-if)#ip address 10.0.0.1 255.0.0.0
RIP(config-if)#no sh

RIP(config-if)#interface se 0/0/1
RIP(config-if)#ip address 51.0.0.1 255.0.0.0
RIP(config-if)#no sh

RIP(config-if)#ip dhcp pool RIP
RIP(dhcp-config)#network 10.0.0.0 255.0.0.0
RIP(dhcp-config)#default-router 10.0.0.1

RIP(config)#router rip
RIP(config-router)#network 10.0.0.0
RIP(config-router)#network 51.0.0.0
