Dhaka Router

Router>enable
Router#configure terminal
Router(config)#hostname Dhaka

Dhaka(config)#interface gig 0/0
Dhaka(config-if)#ip address 192.168.1.1 255.255.255.0
Dhaka(config-if)#no shutdown

Dhaka(config-if)#interface gig 0/1
Dhaka(config-if)#ip address 193.168.1.2 255.255.255.0
Dhaka(config-if)#no shutdown

Dhaka(config-if)#exit

Dhaka(config)#router rip
Dhaka(config-router)#network 192.168.1.0
Dhaka(config-router)#network 193.168.1.0
Dhaka(config-router)#exit
Dhaka(config)#do wr

Dhaka(config)#ip access-list standard Dak-acl
Dhaka(config-std-nacl)#deny 192.168.1.0 0.0.0.63
Dhaka(config-std-nacl)#permit 192.168.1.64 0.0.0.63
Dhaka(config-std-nacl)#deny 192.168.1.128 0.0.0.63
Dhaka(config-std-nacl)#permit 192.168.1.192 0.0.0.63

Dhaka(config-std-nacl)#interface gig 0/0
Dhaka(config-if)#ip access-group Dak-acl in
Dhaka(config-if)#do wr

Dhaka(config)#ip access-list standard chattogram
Dhaka(config-std-nacl)#deny 200.168.1.0 0.0.0.255
Dhaka(config-std-nacl)#permit any


Dhaka(config-std-nacl)#interface gig 0/1
Dhaka(config-if)#ip access-group chattogram in
Dhaka(config-if)#do write
